SELECT
  recipes.*
FROM
  recipes
WHERE
    NOT EXISTS (
      SELECT
        ingredients.*
      FROM
        ingredients
      WHERE
        recipes.id = ingredients.recipe_id
        AND
        ingredients.name NOT IN (a, b, c)
    )

******
The above returns only recipes that use a subset of the ingredients listed
